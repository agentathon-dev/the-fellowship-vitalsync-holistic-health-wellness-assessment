## Description

VitalSync is a comprehensive, evidence-based health and wellness assessment engine that generates personalized, multi-dimensional wellness reports. Unlike simple BMI calculators or generic fitness advice, VitalSync integrates seven interconnected health dimensions into a unified assessment: (1) Physical Health Metrics — calculates BMI, Basal Metabolic Rate (Mifflin-St Jeor equation), Total Daily Energy Expenditure, body fat estimation (U.S. Navy method), and ideal weight ranges using Devine, Robinson, and Miller formulas. (2) Cardiovascular Risk Assessment — a simplified Framingham-inspired scoring model that evaluates age, BMI, smoking status, family history, activity level, diabetes, and hypertension to produce risk stratification with appropriate medical referral triggers. (3) Mental Wellness Screening — structured questionnaires for stress, sleep quality, and mood that use weighted scoring algorithms inspired by validated clinical instruments (PHQ-2 pattern), with automatic severity flagging and crisis resource provision when critical thresholds are met. (4) Personalized Fitness Plans — generates goal-specific periodized training programs (weight loss, muscle gain, general health) with heart rate zone calculations (Tanaka formula), safety modifications based on age and BMI, and progressive overload guidance. (5) Nutrition Planning — calculates individualized macronutrient targets, meal timing strategies, hydration goals, and micronutrient focus areas with appropriate caloric adjustments for user goals. (6) Sleep Quality Assessment — evaluates sleep patterns and provides evidence-based sleep hygiene recommendations including CBT-I referral triggers. (7) Stress Management Protocol — provides structured breathing techniques (4-7-8), grounding exercises (5-4-3-2-1), progressive muscle relaxation, and mindfulness practices with lifestyle modification strategies. A critical safety feature is the Red Flag Detection System that monitors for dangerous BMI extremes, elevated mood distress scores, age-specific exercise contraindications, and cardiovascular risk levels — automatically escalating to crisis resources (988 Lifeline, Crisis Text Line) when warranted. The composite Wellness Score aggregates physical (30%), mental (30%), recovery (25%), and nutrition (15%) dimensions into an actionable 0-100 score with visual progress bars and prioritized action items. Every section includes prominent health disclaimers emphasizing this is educational only and not a substitute for professional medical advice.

/**
 * Calculate Body Mass Index
 */
function calculateBMI(weightKg, heightCm) {
  const heightM = heightCm / 100;
  const bmi = weightKg / (heightM * heightM);
  const category = BMI_CATEGORIES.find((c) => bmi < c.max) || BMI_CATEGORIES[BMI_CATEGORIES.length - 1];
  return { value: Math.round(bmi * 10) / 10, ...category };
}

/**
 * Calculate Basal Metabolic Rate using Mifflin-St Jeor equation
 * (most accurate for general population per ADA)
 */
function calculateBMR(weightKg, heightCm, age, sex) {
  const base = 10 * weightKg + 6.25 * heightCm - 5 * age;
  return sex === "male" ? base + 5 : base - 161;
}

/**
 * Calculate Total Daily Energy Expenditure
 */
function calculateTDEE(bmr, activityLevel) {
  const multiplier = ACTIVITY_MULTIPLIERS[activityLevel]?.factor || 1.2;
  return Math.round(bmr * multiplier);
}

/**
 * Estimate body fat percentage using the U.S. Navy method
 */
function estimateBodyFat(sex, waistCm, neckCm, heightCm, hipCm) {
  if (sex === "male") {
    return Math.round(
      (495 / (1.0324 - 0.19077 * Math.log10(waistCm - neckCm) + 0.15456 * Math.log10(heightCm)) - 450) * 10
    ) / 10;
  } else {
    return Math.round(
      (495 / (1.29579 - 0.35004 * Math.log10(waistCm + hipCm - neckCm) + 0.22100 * Math.log10(heightCm)) - 450) * 10
    ) / 10;
  }
}

/**
 * Calculate ideal weight range using multiple formulas
 */
function idealWeightRange(heightCm, sex) {
  const heightIn = heightCm / 2.54;
  const over60 = Math.max(0, heightIn - 60);
  // Devine formula
  const devine = sex === "male" ? 50 + 2.3 * over60 : 45.5 + 2.3 * over60;
  // Robinson formula
  const robinson = sex === "male" ? 52 + 1.9 * over60 : 49 + 1.7 * over60;
  // Miller formula
  const miller = sex === "male" ? 56.2 + 1.41 * over60 : 53.1 + 1.36 * over60;
  const low = Math.round(Math.min(devine, robinson, miller) * 10) / 10;
  const high = Math.round(Math.max(devine, robinson, miller) * 10) / 10;
  return { low, high, devine: Math.round(devine * 10) / 10 };
}

/**
 * Estimate maximum heart rate and training zones
 */
function heartRateZones(age) {
  const maxHR = 220 - age; // Simplified; Tanaka: 208 - 0.7*age
  const tanaka = Math.round(208 - 0.7 * age);
  return {
    maxHR,
    tanakaMaxHR: tanaka,
    zones: {
      zone1_recovery: { min: Math.round(tanaka * 0.5), max: Math.round(tanaka * 0.6), label: "Recovery (50-60%)" },
      zone2_fatBurn: { min: Math.round(tanaka * 0.6), max: Math.round(tanaka * 0.7), label: "Fat Burn (60-70%)" },
      zone3_aerobic: { min: Math.round(tanaka * 0.7), max: Math.round(tanaka * 0.8), label: "Aerobic (70-80%)" },
      zone4_threshold: { min: Math.round(tanaka * 0.8), max: Math.round(tanaka * 0.9), label: "Threshold (80-90%)" },
      zone5_peak: { min: Math.round(tanaka * 0.9), max: tanaka, label: "Peak (90-100%)" },
    },
  };
}

/**
 * Cardiovascular risk score (simplified, inspired by Framingham)
 * NOT a clinical tool — for educational illustration only
 */
function cardiovascularRiskEstimate(profile) {
  let score = 0;
  // Age factor
  if (profile.age > 55) score += 3;
  else if (profile.age > 45) score += 2;
  else if (profile.age > 35) score += 1;
  // BMI factor
  if (profile.bmi > 35) score += 3;
  else if (profile.bmi > 30) score += 2;
  else if (profile.bmi > 25) score += 1;
  // Smoking
  if (profile.smoker) score += 3;
  // Family history
  if (profile.familyHeartDisease) score += 2;
  // Activity level
  if (profile.activityLevel === "sedentary") score += 2;
  else if (profile.activityLevel === "light") score += 1;
  // Diabetes
  if (profile.diabetes) score += 2;
  // Hypertension
  if (profile.hypertension) score += 2;

  let riskLevel, recommendation;
  if (score <= 2) {
    riskLevel = "Low";
    recommendation = "Maintain healthy lifestyle. Regular check-ups recommended.";
  } else if (score <= 5) {
    riskLevel = "Moderate";
    recommendation = "Consider lifestyle modifications. Consult your doctor for a comprehensive cardiovascular assessment.";
  } else if (score <= 8) {
    riskLevel = "Elevated";
    recommendation = "⚠️ IMPORTANT: Please schedule an appointment with your healthcare provider for a thorough cardiovascular evaluation.";
  } else {
    riskLevel = "High";
    recommendation = "🚨 URGENT: Please consult a healthcare provider as soon as possible for comprehensive cardiovascular risk assessment and management.";
  }
  return { score, maxScore: 18, riskLevel, recommendation };
}

function assessWellnessDomain(questions, responses) {
  let totalScore = 0;
  let maxScore = 0;
  const details = [];
  for (const q of questions) {
    const response = responses[q.id] || 0; // 0-3 scale
    const weighted = response * q.weight;
    totalScore += weighted;
    maxScore += 3 * q.weight;
    if (response >= 2) {
      details.push({ question: q.text, severity: response === 3 ? "frequent" : "common" });
    }
  }
  const percentage = Math.round((totalScore / maxScore) * 100);
  let level, advice;
  if (percentage <= 20) {
    level = "Excellent";
    advice = "Your responses suggest healthy functioning in this area. Keep up your current practices.";
  } else if (percentage <= 40) {
    level = "Good";
    advice = "Minor areas for improvement. Consider small adjustments to optimize this area.";
  } else if (percentage <= 60) {
    level = "Moderate Concern";
    advice = "Some areas need attention. Consider implementing the recommended strategies below.";
  } else if (percentage <= 80) {
    level = "Significant Concern";
    advice = "⚠️ Multiple indicators suggest this area needs focused attention. Consider speaking with a healthcare professional.";
  } else {
    level = "Critical Concern";
    advice = "🚨 Your responses indicate significant difficulty. PLEASE consult a healthcare professional for proper evaluation and support.";
  }
  return { totalScore, maxScore, percentage, level, advice, flaggedItems: details };
}

function generateNutritionPlan(profile) {
  const { tdee, goal, weightKg, activityLevel } = profile;
  let targetCalories, proteinG, fatG, carbG;

  if (goal === "weight-loss") {
    targetCalories = Math.round(tdee * 0.8); // 20% deficit
    proteinG = Math.round(weightKg * 2.0); // high protein to preserve muscle
    fatG = Math.round((targetCalories * 0.25) / 9);
    carbG = Math.round((targetCalories - proteinG * 4 - fatG * 9) / 4);
  } else if (goal === "muscle-gain") {
    targetCalories = Math.round(tdee * 1.1); // 10% surplus
    proteinG = Math.round(weightKg * 2.2);
    fatG = Math.round((targetCalories * 0.25) / 9);
    carbG = Math.round((targetCalories - proteinG * 4 - fatG * 9) / 4);
  } else {
    targetCalories = tdee;
    proteinG = Math.round(weightKg * 1.6);
    fatG = Math.round((targetCalories * 0.3) / 9);
    carbG = Math.round((targetCalories - proteinG * 4 - fatG * 9) / 4);
  }

  const waterLiters = Math.round((weightKg * 0.033 + (activityLevel === "active" || activityLevel === "extreme" ? 0.5 : 0)) * 10) / 10;

  const mealTiming = [
    { meal: "Breakfast (7-8 AM)", macros: `~${Math.round(targetCalories * 0.25)} kcal`, example: "Oatmeal with berries & nuts, or eggs with whole-grain toast and avocado" },
    { meal: "Mid-Morning Snack (10 AM)", macros: `~${Math.round(targetCalories * 0.1)} kcal`, example: "Greek yogurt with fruit, or handful of nuts with an apple" },
    { meal: "Lunch (12-1 PM)", macros: `~${Math.round(targetCalories * 0.3)} kcal`, example: "Lean protein with vegetables and complex carbs (quinoa, sweet potato)" },
    { meal: "Afternoon Snack (3-4 PM)", macros: `~${Math.round(targetCalories * 0.1)} kcal`, example: "Protein shake, or hummus with vegetables" },
    { meal: "Dinner (6-7 PM)", macros: `~${Math.round(targetCalories * 0.25)} kcal`, example: "Grilled fish/chicken with roasted vegetables and brown rice" },

  const micronutrientFocus = [
    "Vitamin D: Fatty fish, fortified foods, sunlight exposure (15-20 min/day)",
    "Omega-3: Salmon, sardines, walnuts, flaxseed (2-3 servings fish/week)",
    "Fiber: 25-35g/day from vegetables, fruits, whole grains, legumes",
    "Calcium: Dairy, leafy greens, fortified alternatives (1000-1200mg/day)",
    "Iron: Lean red meat, spinach, lentils (pair with vitamin C for absorption)",
    "Magnesium: Nuts, seeds, dark chocolate, leafy greens",
  ];

  return {
    targetCalories,
    macros: {
      protein: { grams: proteinG, calories: proteinG * 4, percentage: Math.round((proteinG * 4 / targetCalories) * 100) },
      fat: { grams: fatG, calories: fatG * 9, percentage: Math.round((fatG * 9 / targetCalories) * 100) },

    micronutrientFocus,
    disclaimer: "⚠️ These are general guidelines. Individual needs vary. Consult a registered dietitian for personalized nutrition advice, especially if you have food allergies, intolerances, or medical conditions.",
  };
}

// RED FLAG DETECTION — Critical safety feature

function detectRedFlags(profile, assessments) {
  const flags = [];

  // BMI extremes
  if (profile.bmi < 16) {
    flags.push({ severity: "CRITICAL", message: "BMI is severely low. This may indicate malnutrition or an eating disorder. Please seek immediate medical attention." });
  }
  if (profile.bmi > 40) {
    flags.push({ severity: "HIGH", message: "BMI indicates Class III obesity. Medical supervision is recommended before starting any exercise or diet program." });
  }

  // Mood screening critical flags
  if (assessments.mood) {
    const moodPct = assessments.mood.percentage;
    if (moodPct > 75) {
      flags.push({
        severity: "CRITICAL",
        message: "🚨 Your mood assessment responses indicate significant distress. PLEASE reach out to a mental health professional or crisis service:\n" +
          "  • National Suicide Prevention Lifeline: 988 (call or text)\n" +
          "  • Crisis Text Line: Text HOME to 741741\n" +
          "  • International Association for Suicide Prevention: https://www.iasp.info/resources/Crisis_Centres/",
      });
    } else if (moodPct > 50) {
      flags.push({ severity: "HIGH", message: "Your mood responses suggest you may benefit from speaking with a mental health professional. Please consider scheduling an appointment." });
    }
  }

  // Age + high intensity combination
  if (profile.age > 65 && profile.goal === "muscle-gain") {
    flags.push({ severity: "MODERATE", message: "For individuals over 65, please get medical clearance before starting a strength training program." });
  }

  // Cardiovascular risk
  if (profile.cvRisk && profile.cvRisk.score > 8) {
    flags.push({ severity: "HIGH", message: "Elevated cardiovascular risk factors detected. Please consult a cardiologist before beginning exercise." });
  }

  return flags;
}

// WELLNESS SCORE — Composite multi-dimensional score
// ============================================================================

function calculateWellnessScore(profile, assessments) {
  const dimensions = {};

  // Physical dimension (0-100)
  let physical = 70; // base
  if (profile.bmiCategory.risk === "minimal") physical += 15;
  else if (profile.bmiCategory.risk === "low" || profile.bmiCategory.risk === "moderate") physical += 5;
  else physical -= 10;
  if (profile.activityLevel === "moderate" || profile.activityLevel === "active") physical += 15;
  else if (profile.activityLevel === "sedentary") physical -= 15;
  dimensions.physical = Math.max(0, Math.min(100, physical));

  // Mental dimension
  const moodScore = assessments.mood ? 100 - assessments.mood.percentage : 70;
  const stressScore = assessments.stress ? 100 - assessments.stress.percentage : 70;
  dimensions.mental = Math.round((moodScore + stressScore) / 2);

  // Recovery dimension (sleep quality)
  dimensions.recovery = assessments.sleep ? 100 - assessments.sleep.percentage : 70;

  // Nutrition readiness (based on goal alignment)
  dimensions.nutrition = profile.goal ? 65 : 50; // Having a goal = better baseline

  // Overall composite (weighted)
  const overall = Math.round(
    dimensions.physical * 0.3 +
    dimensions.mental * 0.3 +
    dimensions.recovery * 0.25 +
    dimensions.nutrition * 0.15
  );

  return { overall, dimensions };
}

function generateReport(profile, assessments, plans) {
  const wellnessScore = calculateWellnessScore(profile, assessments);
  const redFlags = detectRedFlags(profile, assessments);

  const report = {
    header: {
      title: "═══════════════ VitalSync Wellness Assessment Report ═══════════════",
      generatedAt: new Date().toISOString(),
      disclaimer: DISCLAIMER,
    },

    redFlags: redFlags.length > 0 ? redFlags : null,

    profileSummary: {
      age: profile.age,
      sex: profile.sex,
      height: `${profile.heightCm} cm`,
      weight: `${profile.weightKg} kg`,
      bmi: `${profile.bmi} (${profile.bmiCategory.label}) ${profile.bmiCategory.color}`,
      bmr: `${profile.bmr} kcal/day`,
      tdee: `${profile.tdee} kcal/day`,
      idealWeightRange: `${profile.idealWeight.low} - ${profile.idealWeight.high} kg`,
      activityLevel: ACTIVITY_MULTIPLIERS[profile.activityLevel]?.label,
      goal: profile.goal,
    },

    wellnessScore: {
      overall: `${wellnessScore.overall}/100`,
      dimensions: wellnessScore.dimensions,
      interpretation: wellnessScore.overall >= 80
        ? "🟢 Excellent — You're doing great! Focus on maintaining these healthy habits."
        : wellnessScore.overall >= 60
        ? "🟡 Good — Solid foundation with room for targeted improvement."
        : wellnessScore.overall >= 40
        ? "🟠 Fair — Several areas would benefit from focused attention."
        : "🔴 Needs Attention — Consider working with healthcare professionals on a comprehensive plan.",
    },

    cardiovascularRisk: profile.cvRisk,

    wellnessAssessments: {
      stress: assessments.stress,
      sleep: assessments.sleep,
      mood: assessments.mood,
    },

    fitnessPlan: plans.fitness,
    nutritionPlan: plans.nutrition,
    sleepRecommendations: plans.sleep,
    stressManagement: plans.stressManagement,

    actionItems: generateActionItems(wellnessScore, assessments, redFlags),

    closing: {
      message: "This report was generated by VitalSync for educational purposes. Track your progress by re-running this assessment monthly.",
      reminder: "⚠️ Always consult qualified healthcare professionals before making significant changes to your exercise routine, diet, or health management approach.",
      emergencyResources: {
        emergency: "Call 911 (US) or your local emergency number",
        suicidePrevention: "988 Suicide & Crisis Lifeline: Call or text 988",
        crisisText: "Crisis Text Line: Text HOME to 741741",
        samhsa: "SAMHSA Helpline: 1-800-662-4357",
      },
    },
  };

  return report;
}

function generateActionItems(wellnessScore, assessments, redFlags) {
  const items = [];
  const dims = wellnessScore.dimensions;

  // Priority: red flags first
  if (redFlags.length > 0) {
    items.push({ priority: "🚨 IMMEDIATE", action: "Address the red flags listed above. Seek appropriate professional guidance." });
  }

  // Lowest scoring dimensions get attention
  const sorted = Object.entries(dims).sort((a, b) => a[1] - b[1]);
  for (const [dim, score] of sorted) {
    if (score < 50) {
      const actions = {
        physical: "Start with 15-minute daily walks and gradually increase activity",
        mental: "Practice one stress management technique daily. Consider professional support.",
        recovery: "Prioritize sleep hygiene — set a consistent bedtime tonight",
        nutrition: "Begin tracking meals for one week to build awareness",
      };
      items.push({ priority: "⬆️ HIGH", action: `${dim}: ${actions[dim] || "Focus on improving this area"}` });
    } else if (score < 70) {
      items.push({ priority: "📋 MODERATE", action: `${dim}: Room for improvement. Follow the specific recommendations in your plan.` });
    }
  }

  items.push({ priority: "📅 ONGOING", action: "Re-assess monthly to track progress. Celebrate improvements!" });
  return items;
}
